import requests import json from urllib.parse import urlencode def test_wfs_connection(api_key): """ NSDI 지적도 WFS 연결 및 데이터 조회 종합 테스트 """ base_url = "https://api.vworld.kr/req/wfs" # 테스트 시나리오 정의 test_cases = { "1. GetCapabilities": { "params": { "SERVICE": "WFS", "VERSION": "2.0.0", "REQUEST": "GetCapabilities", "key": api_key }, "expected_content": "ServiceIdentification" }, "2. DescribeFeatureType": { "params": { "SERVICE": "WFS", "VERSION": "2.0.0", "REQUEST": "DescribeFeatureType", "TYPENAME": "lp_pa_cbnd_bubun", "key": api_key }, "expected_content": "complexType" }, "3. GetFeature (WGS84)": { "params": { "SERVICE": "WFS", "VERSION": "2.0.0", "REQUEST": "GetFeature", "TYPENAME": "lp_pa_cbnd_bubun", "SRSNAME": "EPSG:4326", "BBOX": "126.974,37.575,126.977,37.578,EPSG:4326", "COUNT": "5", "OUTPUTFORMAT": "application/json", "key": api_key }, "expected_content": "features" }, "4. GetFeature (Korea 2000)": { "params": { "SERVICE": "WFS", "VERSION": "2.0.0", "REQUEST": "GetFeature", "TYPENAME": "lp_pa_cbnd_bubun", "SRSNAME": "EPSG:5179", "BBOX": "953900,1952100,954200,1952400,EPSG:5179", "COUNT": "5", "OUTPUTFORMAT": "application/json", "key": api_key }, "expected_content": "features" }, "5. CQL Filter Test": { "params": { "SERVICE": "WFS", "VERSION": "2.0.0", "REQUEST": "GetFeature", "TYPENAME": "lp_pa_cbnd_bubun", "SRSNAME": "EPSG:4326", "CQL_FILTER": "jimok='대'", "COUNT": "3", "OUTPUTFORMAT": "application/json", "key": api_key }, "expected_content": "features" } } print("🚀 NSDI 지적도 WFS 연결 테스트 시작") print("=" * 60) success_count = 0 for test_name, config in test_cases.items(): print(f"\n{test_name}") print("-" * 40) # URL 생성 full_url = f"{base_url}?{urlencode(config['params'])}" print(f"🔗 URL: {full_url}") try: response = requests.get(base_url, params=config["params"], timeout=30) # 상태 코드 확인 print(f"📊 HTTP Status: {response.status_code}") print(f"📦 Content-Type: {response.headers.get('Content-Type', 'N/A')}") print(f"📏 Response Size: {len(response.content):,} bytes") if response.status_code == 200: # 응답 내용 검증 content = response.text if config["expected_content"] in content: print(f"✅ 테스트 성공!") success_count += 1 # JSON 응답인 경우 추가 분석 if "application/json" in response.headers.get("Content-Type", ""): try: data = response.json() if "features" in data: features = data["features"] print(f"🏠 반환된 필지 수: {len(features)}개") if features: first_feature = features[0] props = first_feature.get("properties", {}) print(f"📋 첫 번째 필지 속성:") for key, value in list(props.items())[:5]: print(f" • {key}: {value}") except json.JSONDecodeError: print("⚠️ JSON 파싱 실패") else: print(f"❌ 예상 내용 '{config['expected_content']}' 없음") print(f"📝 응답 미리보기: {content[:200]}...") else: print(f"❌ HTTP 오류: {response.status_code}") print(f"📝 오류 내용: {response.text[:300]}...") except requests.exceptions.Timeout: print("⏰ 요청 시간 초과 (30초)") except requests.exceptions.ConnectionError: print("🔌 연결 오류 - 네트워크 상태 확인") except Exception as e: print(f"❌ 예외 발생: {e}") print(f"\n{'='*60}") print(f"🎯 테스트 완료: {success_count}/{len(test_cases)} 성공") if success_count == len(test_cases): print("🎉 모든 테스트 통과! WFS 서비스가 정상 작동합니다.") elif success_count > 0: print("⚠️ 일부 테스트만 성공. API 키나 파라미터를 확인하세요.") else: print("💥 모든 테스트 실패. API 키나 네트워크 상태를 점검하세요.") # 실행 예시 if __name__ == "__main__": API_KEY = "여기에_실제_API_키_입력" test_wfs_connection(API_KEY)
